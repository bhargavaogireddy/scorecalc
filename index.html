<script>

    let players = [];

    let rounds = [];


    function generateTournament() {

        const rawInput = document.getElementById('player-input').value;

        // Split by lines and remove empty values

        const names = rawInput.split('\n').map(n => n.trim()).filter(n => n !== "");


        if (names.length < 2) {

            alert("Please enter at least 2 players.");

            return;

        }


        // Initialize players array

        players = names.map((name, index) => ({

            id: index,

            name: name,

            wins: 0,

            losses: 0,

            pointsFor: 0,

            pointsAgainst: 0,

            differential: 0

        }));


        // Clone list to manipulate for Round Robin algorithm (Circle Method)

        let list = [...players];

        

        // If odd, add a dummy 'BYE' player

        const isOdd = list.length % 2 !== 0;

        if (isOdd) {

            list.push({ id: -1, name: 'BYE' });

        }


        const numPlayers = list.length;

        const numRounds = numPlayers - 1;

        const matchesPerRound = numPlayers / 2;


        rounds = [];


        // Generate Round Robin pairings using Circle Method

        for (let r = 0; r < numRounds; r++) {

            let roundMatches = [];

            for (let m = 0; m < matchesPerRound; m++) {

                let p1 = list[m];

                let p2 = list[numPlayers - 1 - m];


                // Skip matches involving the BYE dummy player

                if (p1.id !== -1 && p2.id !== -1) {

                    roundMatches.push({ p1: p1, p2: p2 });

                }

            }

            rounds.push(roundMatches);


            // Rotate list elements (leave first element fixed)

            list.splice(1, 0, list.pop());

        }


        // Shuffle rounds to randomize the order of play

        rounds.sort(() => Math.random() - 0.5);


        renderRounds();

        

        document.getElementById('setup-stage').classList.add('hidden');

        document.getElementById('tournament-stage').classList.remove('hidden');

    }


    function renderRounds() {

        const container = document.getElementById('rounds-container');

        container.innerHTML = '';


        rounds.forEach((round, roundIndex) => {

            const roundDiv = document.createElement('div');

            roundDiv.className = 'round-block';

            

            const title = document.createElement('div');

            title.className = 'round-title';

            title.innerText = `Round ${roundIndex + 1}`;

            roundDiv.appendChild(title);


            round.forEach((match, matchIndex) => {

                const matchId = `r${roundIndex}m${matchIndex}`;

                

                const matchupDiv = document.createElement('div');

                matchupDiv.className = 'matchup';

                matchupDiv.innerHTML = `

                    <span class="player-name">${match.p1.name}</span>

                    <input type="number" id="${matchId}-s1" placeholder="Score" min="0">

                    <input type="number" id="${matchId}-s2" placeholder="Score" min="0">

                    <span class="player-name p2">${match.p2.name}</span>

                `;

                roundDiv.appendChild(matchupDiv);

            });


            container.appendChild(roundDiv);

        });

    }


    function calculateStandings() {

        // Reset player metrics before tabulating

        players.forEach(p => {

            p.wins = 0;

            p.losses = 0;

            p.pointsFor = 0;

            p.pointsAgainst = 0;

            p.differential = 0;

        });


        // Loop through inputs and tabulate results

        for (let r = 0; r < rounds.length; r++) {

            for (let m = 0; m < rounds[r].length; m++) {

                const matchId = `r${r}m${m}`;

                const s1Input = document.getElementById(`${matchId}-s1`).value;

                const s2Input = document.getElementById(`${matchId}-s2`).value;


                // Validation check to make sure scores are filled out

                if (s1Input === "" || s2Input === "") {

                    alert("Please fill out all match scores before finalizing standings.");

                    return;

                }


                const s1 = parseInt(s1Input, 10);

                const s2 = parseInt(s2Input, 10);


                let p1 = players.find(p => p.id === rounds[r][m].p1.id);

                let p2 = players.find(p => p.id === rounds[r][m].p2.id);


                p1.pointsFor += s1;

                p1.pointsAgainst += s2;

                p2.pointsFor += s2;

                p2.pointsAgainst += s1;


                if (s1 > s2) {

                    p1.wins += 1;

                    p2.losses += 1;

                } else if (s2 > s1) {

                    p2.wins += 1;

                    p1.losses += 1;

                } else {

                    // Tie scenario handling (adjust if your sport doesn't allow ties)

                    p1.wins += 0.5;

                    p1.losses += 0.5;

                    p2.wins += 0.5;

                    p2.losses += 0.5;

                }

            }

        }


        // Post-calculate differentials

        players.forEach(p => {

            p.differential = p.pointsFor - p.pointsAgainst;

        });


        // TIE BREAKER SORTING LOGIC:

        // 1. Wins (Descending)

        // 2. Point Differential (Descending)

        // 3. Total Points For (Descending)

        players.sort((a, b) => {

            if (b.wins !== a.wins) {

                return b.wins - a.wins;

            }

            if (b.differential !== a.differential) {

                return b.differential - a.differential;

            }

            return b.pointsFor - a.pointsFor;

        });


        renderStandings();

        document.getElementById('standings-stage').classList.remove('hidden');

        // Scroll down to see results

        document.getElementById('standings-stage').scrollIntoView({ behavior: 'smooth' });

    }


    function renderStandings() {

        const tbody = document.querySelector('#standings-table tbody');

        tbody.innerHTML = '';


        players.forEach((player, index) => {

            const tr = document.createElement('tr');

            tr.innerHTML = `

                <td><strong>#${index + 1}</strong></td>

                <td>${player.name}</td>

                <td>${player.wins} - ${player.losses}</td>

                <td>${player.pointsFor}</td>

                <td>${player.pointsAgainst}</td>

                <td>${player.differential > 0 ? '+' : ''}${player.differential}</td>

            `;

            tbody.appendChild(tr);

        });

    }


    function resetTournament() {

        if(confirm("Are you sure you want to reset? This clears all current match scores.")) {

            document.getElementById('standings-stage').classList.add('hidden');

            document.getElementById('tournament-stage').classList.add('hidden');

            document.getElementById('setup-stage').classList.remove('hidden');

            document.getElementById('player-input').value = '';

        }

    }
